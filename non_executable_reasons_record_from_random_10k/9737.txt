c_range = np.arange(0,3.01,0.01)

def x_filler(c_i):
    eq = lambda x : (1-np.exp((-c_i*x))-x)
    eq_sol = optimize.fsolve(eq,1.,xtol=1e-7)

plt.ylabel(r'$x(c)$')
